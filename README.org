  (setq inhibit-startup-screen t)                 ;; Disable the startup screen

** IDE initialization layout

This is an implementation of an IDE look alike for emacs, if you are opening emacs normally it only shows sratchpad buffer
but if you are opening a project via: *"folder -> right click -> open with emacs"* it opens with the following layout

#+begin_src emacs-lisp
  ;;--- IDE (like) initialization layout ---
  (add-hook 'emacs-startup-hook
            (lambda ()
  	    (toggle-frame-maximized)
  	    ;; Check if command line arguments were passed and if the first one is a directory
              (when (and command-line-args
                         (let ((first-arg (car (last command-line-args))))
  			 (and first-arg (file-directory-p first-arg))))
                (multi-vterm-dedicated-open)
                (other-window -1)
  	      ;; Dynamically calculate screen height and adjust vterm 
  	      (let ((vterm-lines (- (round (* (frame-height) 0.25)) (window-height (next-window)))))
  		(window-resize (next-window) vterm-lines))
                ;;(enlarge-window 20)
                (neotree-toggle))))

#+end_src

* Emacs Repositories
This section is to configure and install software from other package managers that emacs have.
** MELPA
*** Setup
#+begin_src emacs-lisp
  ;;--- MELPA setup ---
  (require 'package)
  (let* ((no-ssl (and (memq system-type '(windows-nt ms-dos))
                      (not (gnutls-available-p))))
         (proto (if no-ssl "http" "https")))
    (when no-ssl
      (warn "SSL not supported. Consider upgrading Emacs for secure connections."))
    (add-to-list 'package-archives (cons "melpa" (concat proto "://melpa.org/packages/")) t))
  (package-initialize)
#+end_src

* Emacs and packages general configuration
** Neotree
This configures neotree to appear with F12, its size, and prevent showing dotted files

#+begin_src emacs-lisp
  ;;--- Installed packages configurations ---
  (use-package neotree
    :ensure t
    :bind ([f12] . neotree-toggle)
    :config
    (setq neo-theme (if (display-graphic-p) 'icons 'arrow))
    (setq neo-window-width 40)
    (with-eval-after-load 'neotree
      (setq neo-hidden-regexp-list '("^\\." "__pycache__" "~$" "^#.*#$"))
      )
    )
#+end_src
